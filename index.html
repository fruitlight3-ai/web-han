# 1. Menampilkan menu pilihan kepada pengguna print("=== Kalkulator Sederhana ===") print("Pilih operasi matematika:") print("1. Penjumlahan (+)") print("2. Pengurangan (-)") print("3. Perkalian (*)") print("4. Pembagian (/)") # 2. Mengambil input pilihan operasi pilihan = input("Masukkan pilihan (1/2/3/4): ") # 3. Memastikan input yang dimasukkan valid if pilihan in ['1', '2', '3', '4']: # Mengambil input angka dari pengguna # Menggunakan float() agar bisa menghitung angka desimal angka1 = float(input("Masukkan angka pertama: ")) angka2 = float(input("Masukkan angka kedua: ")) # 4. Melakukan perhitungan berdasarkan pilihan menggunakan percabangan if pilihan == '1': hasil = angka1 + angka2 print("Hasil:", angka1, "+", angka2, "=", hasil) elif pilihan == '2': hasil = angka1 - angka2 print("Hasil:", angka1, "-", angka2, "=", hasil) elif pilihan == '3': hasil = angka1 * angka2 print("Hasil:", angka1, "*", angka2, "=", hasil) elif pilihan == '4': # Mencegah error pembagian dengan nol if angka2 == 0: print("Error: Tidak bisa melakukan pembagian dengan angka 0!") else: hasil = angka1 / angka2 print("Hasil:", angka1, "/", angka2, "=", hasil) else: # Pesan jika pengguna memasukkan selain angka 1, 2, 3, atau 4 print("Input tidak valid! Harap masukkan angka 1, 2, 3, atau 4.")
